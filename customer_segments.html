import random
from sklearn.model_selection import train_test_split
from sklearn.tree import DecisionTreeRegressor
import itertools
def choices(l, k=1):
    ret = []
    for i in range(k):
        res = random.choice(l)
        ret.append(res)
        l.remove(res)
    return ret

random.choices = choices

model = dict()
for target_name in data.columns:
    for feature_count in range(2, len(data.columns)):
        column_left = list(set(data.columns) - {target_name})
        features_pair = itertools.permutations(column_left, feature_count)
        
        for features in features_pair:        
            record = tuple(features) + (target_name, )
            new_data = data[list(features)]
            target = data[target_name]

            # TODO

            score = regressor.score(X_test, y_test)

            res[record] = score
