import random
from sklearn.model_selection import train_test_split
from sklearn.tree import DecisionTreeRegressor
import itertools
def choices(l, k=1):
    ret = []
    for i in range(k):
        res = random.choice(l)
        ret.append(res)
        l.remove(res)
    return ret

random.choices = choices

res = {}
model = {}
for feature in data.columns:
    new_data = data.drop([feature], axis=1)
    target = data[feature]
    
    X_train, X_test, y_train, y_test = train_test_split(new_data, target, test_size=0.3, random_state=233)
    
    regressor = DecisionTreeRegressor(random_state=233)
    regressor.fit(X_train, y_train)
    
    score = regressor.score(X_test, y_test)
    
    key = ' '.join(new_data.columns)
    res[key] = score
    model[key] = regressor
